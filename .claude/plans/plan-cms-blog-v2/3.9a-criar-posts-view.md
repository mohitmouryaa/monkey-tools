# ✅ 3.9a — Criar view `posts-view.tsx` (listagem)

## Contexto
View que renderiza a lista de posts via `PostsList` + `PostsListItem` dedicados (3.2a) e usa `useRemovePost` para deleção.

## Detalhe técnico
Criar `apps/web/modules/dashboard/ui/views/posts-view.tsx`:

```tsx
"use client";

import { Newspaper } from "lucide-react";
import { useRouter } from "next/navigation";
import { useSuspensePosts } from "@/modules/dashboard/hooks/use-suspense-posts";
import { useRemovePost } from "@/modules/dashboard/hooks/use-remove-post";
import { PostsList, PostsListItem, PostsEmpty } from "@/modules/dashboard/posts/ui/posts-entity-components";

export const PostsView = () => {
  const posts = useSuspensePosts();

  return (
    <PostsList
      items={posts.data.items}
      getKey={(p: { _id: string }) => p._id}
      renderItem={(p) => <PostItem data={p} />}
      emptyView={<PostsEmptySection />}
    />
  );
};

const PostsEmptySection = () => {
  const router = useRouter();
  return <PostsEmpty message="No posts yet. Create your first post." onNew={() => router.push("/dashboard/posts/new")} />;
};

const PostItem = ({ data }: { data: { _id: string; title: string; status: string; slug: string } }) => {
  const remove = useRemovePost();
  const handleRemove = () => {
    if (confirm(`Delete post "${data.title}"?`)) {
      remove.mutate({ id: data._id });
    }
  };
  return (
    <PostsListItem
      href={`/dashboard/posts/${data._id}`}
      title={data.title}
      subtitle={`/${data.slug} · ${data.status}`}
      image={
        <div className="flex items-center justify-center size-8">
          <Newspaper className="size-5" />
        </div>
      }
      onRemove={handleRemove}
      isRemoving={remove.isPending}
    />
  );
};
```

(Tipos: importar `Post` de `@workspace/database` se útil; o que importa é alinhar com o shape que `posts.list` retorna do servidor — `{ items: Array<{ _id, title, slug, status, ... }> }`.)
