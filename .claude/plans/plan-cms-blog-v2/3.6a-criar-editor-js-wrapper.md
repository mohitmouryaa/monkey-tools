# 3.6a — Criar wrapper React do Editor.js ✅

## Contexto
Editor.js é client-only e DOM imperativo. Wrapper monta a instância em `useEffect`, expõe `value: object` (JSON) e `onChange`. Importado pelo form via `next/dynamic({ ssr: false })`.

## Detalhe técnico
Criar `apps/web/modules/dashboard/ui/components/editor-js/index.tsx`:

```tsx
"use client";

import EditorJS, { type OutputData } from "@editorjs/editorjs";
import Header from "@editorjs/header";
import List from "@editorjs/list";
import Quote from "@editorjs/quote";
import Embed from "@editorjs/embed";
import Table from "@editorjs/table";
import Checklist from "@editorjs/checklist";
import ImageTool from "@editorjs/image";
import { useEffect, useRef } from "react";
import { useQueryClient } from "@tanstack/react-query";
import { usePostFileUpload } from "@/modules/dashboard/hooks/use-post-file-upload";
import { ToolEmbedBlock } from "@/modules/dashboard/ui/components/editor-js/tool-embed-block";

interface EditorJsProps {
  value?: OutputData | object;
  onChange: (data: OutputData) => void;
  holderId?: string;
}

export const EditorJsWrapper = ({ value, onChange, holderId = "post-editor-js" }: EditorJsProps) => {
  const ref = useRef<EditorJS | null>(null);
  const queryClient = useQueryClient();
  const { uploadFile } = usePostFileUpload();

  useEffect(() => {
    if (ref.current) return; // already initialized

    const editor = new EditorJS({
      holder: holderId,
      data: (value as OutputData) ?? { blocks: [] },
      tools: {
        header: { class: Header, inlineToolbar: true },
        list: { class: List, inlineToolbar: true },
        quote: { class: Quote, inlineToolbar: true },
        embed: Embed,
        table: Table,
        checklist: Checklist,
        image: {
          class: ImageTool,
          config: {
            uploader: {
              uploadByFile: async (file: File) => {
                const { url } = await uploadFile(file);
                return { success: 1, file: { url } };
              },
            },
          },
        },
        toolEmbed: {
          class: ToolEmbedBlock,
          config: { queryClient },
        },
      },
      async onChange(api) {
        const saved = await api.saver.save();
        onChange(saved);
      },
    });

    ref.current = editor;

    return () => {
      // Editor.js destroy is async; ignore errors on hot reload
      ref.current?.isReady
        .then(() => {
          ref.current?.destroy();
          ref.current = null;
        })
        .catch(() => {
          ref.current = null;
        });
    };
    // eslint-disable-next-line react-hooks/exhaustive-deps
  }, []); // initialize once

  return <div id={holderId} className="min-h-[400px] border rounded-lg p-4 bg-background" />;
};

export default EditorJsWrapper;
```

Notas:
- `default export` permite `next/dynamic(() => import(...), { ssr: false })`.
- O consumer (post-form) faz `dynamic(() => import("@/modules/dashboard/ui/components/editor-js"), { ssr: false })`.
- O `ref.current` evita reinicializar (efeito roda 1x; React 19 strict pode rodar 2x em dev — guard previne).
- `ToolEmbedBlock` recebe `queryClient` via `config` para fazer `fetchQuery` dentro do plugin (P7).
